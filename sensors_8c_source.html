142can_msg conv2CANVelocityData(bool vehicle_direction, double relative_velocity, double relative_acceleration)
143{
144 //printf("Rel acel: %lf\n", relative_acceleration);
145 can_msg aux = {.identifier = ID_SPEED_S, .dataFrame = BASE_DATA_FRAME};
146
147 // Vehicle direction (forward or reverse) data encapsulation
148 if (vehicle_direction)
149 {
150 aux.dataFrame[2] = 0x01;
151 }
152 else
153 {
154 aux.dataFrame[2] = 0x00;
155 }
156
157 // Speed data ​​encapsulation
158 unsigned int data_speed = relative_velocity / RES_SPEED_S;
159 unsigned char ms_speed, ls_speed;
160 ls_speed = data_speed;
161 ms_speed = data_speed >> 8;
162
163 // Defines most and least significant bytes, according to the DBC specification
164 aux.dataFrame[0] = ls_speed;
165 aux.dataFrame[1] = ms_speed;
166
167 // Acceleration data ​​encapsulation
168 double aux_acel = relative_acceleration;
169 if(aux_acel < 0){
170 aux_acel *= -1;
171 aux.dataFrame[5] = 0x01;
172 } else {
173 aux.dataFrame[5] = 0x00;
174 }
175
176 unsigned int data_acel = ((aux_acel * RES_ACCELERATION_DIV_S) - OFFSET_ACCELERATION_S);
177 unsigned char ms_acel, ls_acel;
178 ls_acel = data_acel;
179 ms_acel = data_acel >> 8;
180
181 // Defines most and least significant bytes, according to the DBC specification
182 aux.dataFrame[3] = ls_acel;
183 aux.dataFrame[4] = ms_acel;
184
185 return aux;
186}
